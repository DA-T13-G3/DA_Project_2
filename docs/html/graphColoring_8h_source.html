152int graphColoringBasic(Graph<T>* g,unsigned int N) {
153 std::vector<Vertex<T>*>stack;
154 for (auto v:g->getVertexSet()) {
155 v->setVisited(false);
156 v->setIndegree(v->getAdj().size());
157 }
158 int n_nodes=g->getVertexSet().size();
159 bool all_remaining_nodes_degree_eq_gt_N=true;
160 while (n_nodes>0) {
161 for (auto v:g->getVertexSet()) {
162 if (!v->isVisited()) {
163 if (v->getIndegree()<N) {
164 n_nodes--;
165 stack.push_back(v);
166 v->setVisited(true);
167 for (auto e:v->getAdj()) {
168 Vertex<T>* w=e->getDest();
169 w->setIndegree(w->getIndegree()-1);
170 }
171 }
172 }
173 }
174 all_remaining_nodes_degree_eq_gt_N=true;
175 for (auto v:g->getVertexSet()) {
176 if (!v->isVisited()) {
177 if (v->getIndegree()<N) {
178 all_remaining_nodes_degree_eq_gt_N=false;
179 }
180 }
181 }
182 if (n_nodes!=0 && all_remaining_nodes_degree_eq_gt_N) {
183 return -1;
184 }
185 }
186
187 for (auto &v:g->getVertexSet()) {
188 v->setVisited(false);
189 v->setNum(-1);
190 }
191
192
193
194 int biggestColor=-1;
195 std::vector<bool>usedColors(N,0);
196
197 while (!stack.empty()) {
198 Vertex<T>* v=stack.back();
199 stack.pop_back();
200 for (auto e:v->getAdj()) {
201 Vertex<T>*w=e->getDest();
202 if (w->getNum()!=-1) {
203 usedColors[w->getNum()]=true;
204 }
205 }
206
207 for ( int i=0;i<N;i++) {
208 if (!usedColors[i]) {
209 v->setNum(i);
210 biggestColor=std::max(i,biggestColor);
211 break;
212 }
213 }
214 for ( unsigned int i=0;i<N;i++) {
215 usedColors[i] = false;
216 }
217 }
218
